Jacobi method 

from pprint import pprint 
from numpy import array, zeros, diag, diagflat, dot
	
def jacobi(A, b, N=25, x=None):
	if x is None:
		x = zeros(len(A[0]))
	D = diag(A)
	R = A - diagflat(D)
		
	for i in range(N):
		x = (b - dot(R, x)) / D
	return x
	
A = array([[9,4,1],[1,6,0],[1,-2,-6]])
b = array([-17, 4, 14])
guess = array([0,0,0])
	
sol = jacobi(A,b,N=25, x=guess)
	
print('A:')
pprint(A)
	
print('b:')
pprint(b)
	
print('x:')
pprint(sol)
