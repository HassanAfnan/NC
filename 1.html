Euler method 

	 
import numpy as np 

import math as mt 

def integrate(F,x,y,xStop,h): 

    
    X= [] 

    Y= [] 

    X.append(x) 

    Y.append(y) 

    while x < xStop: 

        h = min(h,xStop - x) 

        y = y + h*F 

        x = x + h 

        X.append(x) 

        Y.append(y) 

    return np.array(X),np.array(Y) 


x= 0.0 
xStop= 0.5
y=np.array(1)
F=mt.sin(y) -1
h=0.1
print(integrate (F,x,y,xStop,h))
