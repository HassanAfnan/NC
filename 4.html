Newton method

Determine the root of f(x) = x3 - e-2x +1 with a tolerance value of 10-4 
import numpy as np 

import math

def newton(f, fprime, pin, eps, N): 

	n=1 

	p = 0 
	while n <= N:
		p = pin - f(pin)/fprime(pin)
		if np.isclose(f(p), 0) or np.abs(p-pin) < eps: 
			print('p is', p, ' and the iteration number is ', n)
			return 
		pin = p
		n +=1 
	y = f(p)
	print(' the last iteration gives', p, ' with function value ', y )
newton(lambda x: x**3 - math.exp(-2*x) + 1, lambda x: 3*x**2 + 2 * math.exp(-2*x), 1, 10e-4, 20)
